## Controlled

<Canvas>
  <Story
    name="Controlled"
    args={{ expandedSections: [0] }}
    argTypes={{ isCollapsible: { control: false }, isExpandable: { control: false } }}
  >
    {args => {
      const [expandedSections, setExpandedSections] = useState(args.expandedSections);
      const handleChange = index => {
        setExpandedSections(sections =>
          expandedSections.includes(index)
            ? sections.filter(section => section !== index)
            : [...expandedSections, index]
        );
      };
      return (
        <Accordion {...args} expandedSections={expandedSections} onChange={handleChange}>
          {args.sections.map((section, index) => (
            <Accordion.Section key={index}>
              <Accordion.Header>
                <Accordion.Label>{section.headerLabel}</Accordion.Label>
              </Accordion.Header>
              <Accordion.Panel>{section.panel}</Accordion.Panel>
            </Accordion.Section>
          ))}
        </Accordion>
      );
    }}
  </Story>
</Canvas>
